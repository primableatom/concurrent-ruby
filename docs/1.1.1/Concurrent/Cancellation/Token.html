def join(*tokens, &block)
  block ||= -> token_list { Promises.any_event(*token_list.map(&:to_event)) }
  self.class.new block.call([@Cancel, *tokens])
end

def loop_until_canceled(&block)
  until canceled?
    result = block.call
  end
  result
end
